\NewDocumentCommand{\gpp}{ O{} O{\dx} O{\dx[2]} O{x} O{x} O{x} }{%
    \frac{1}{#3}
    \kl{#1 
        \kl{#4 \ifthenelse{\equal{#2}{}}{}{+ #2}} - 
        2 \cdot #1\kl{#5} +
        #1 \kl{#6\ifthenelse{\equal{#2}{}}{}{- #2}}
    }
}
